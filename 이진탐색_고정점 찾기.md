```python
def binary_search(array, start, end):
    while start <= end :
        # 중간 값의 '인덱스'
        mid = (start + end) // 2
        # 중간 값이 인덱스 값과 일치 하는 경우 중간 값 반환
        if array[mid] == mid :
            return mid
        # 중간 값이 중간 값의 인덱스 값보다 크면 왼쪽 확인
        elif array[mid] > mid :
            end = mid - 1
        # 중간 값이 중간 값의 인덱스 값보다 작으면 오른쪽 확인
        else :
            start = mid + 1
    return None

n = int(input())
array = list(map(int, input().split()))

result = binary_search(array, 0, n-1)


if result == None:
    print(-1)
else :
    print(result)
```
